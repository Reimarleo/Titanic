# Add age groups to the data frame

def age_groups(age):
    if 0 < age <= 2:
        return 1
    if 2 < age <= 15:
        return 2
    if 15 < age <= 35:
        return 3
    if 35 < age <= 50:
        return 4
    if 50 < age <= 65:
        return 5
    if 65 < age <= 80: #I know 80 is the maximum age from df.describe()
        return 6
    else:
        return 0
    
df['Age_group'] = df['Age'].apply(age_groups)

### Find which variables are most correlated to survival

Run a correlation analysys on all metric variables


```python
# Calculate Pearson's r
def correlation_with_survival(x):
    survived_series = df.Survived
    #Standardize the Survived column
    #I am setting ddof to 1 (Bessel's correction) because this is a sample and not the population
    std_Survived = (survived_series - survived_series.mean()) / survived_series.std(ddof=1)
    
    try:
        #Standardize the series passed to the function
        std_x = (x - x.mean()) / x.std(ddof=1)
        correlation = (std_x * std_Survived).mean()
        #I am returning the absolute value because I am not interested in the direction at this stage
        #I only want to rank the variables in order of importance
        return abs(correlation)
    except:
        return 'N/A'

# Create a series with all the correlations
correlations = df.apply(correlation_with_survival)

# Show correlations in order of strenght
correlations.sort_values(ascending=False)
```

### Considerations on the result

- Non-metric data has been given the 'N/A' value by the function because it can't be used to calculate correlation
- The correlation of 'Survived' with itself is not 1 because the Bessel's correction was used (ddof=1)
- The full 'Age' series captures the correlation between the two variables better than the arbitrarely defined groups
- Passenger IDs were clearly assigned randomly
- The variables with the highest correlation are sex, social class and ticket fare, although the last two are probably highly correlated
